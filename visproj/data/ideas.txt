def read_tweets(filename):
    tweet_t = []
    user_n = []
    with open(filename,) as csvfile:
        filereader = csv.reader(csvfile, delimiter=',', quotechar='"')
        for row in filereader:
            try:
                tweet_t.append(row[2])
                user_n.append(row[3])
            except IndexError:
                pass
    return tweet_t,user_n
